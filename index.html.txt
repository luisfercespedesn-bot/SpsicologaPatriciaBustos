<script>
  const apiKey = "";

  function toggleOffer() {
    const bubble = document.getElementById('offerBubble');
    bubble.classList.toggle('visible');
  }

  // Se muestra sutilmente después de 5 segundos
  setTimeout(() => {
    document.getElementById('offerBubble').classList.add('visible');
    // Se oculta automáticamente tras 10 segundos para no molestar, a menos que el usuario lo vuelva a abrir
    setTimeout(() => {
        document.getElementById('offerBubble').classList.remove('visible');
    }, 10000);
  }, 5000);

  async function callGemini(promptText) {
    const responseDiv = document.getElementById('ai-response');
    const loading = document.getElementById('loading');
    responseDiv.style.display = 'none';
    loading.style.display = 'block';

    let retries = 0;
    while (retries < 5) {
      try {
        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            contents: [{ parts: [{ text: promptText }] }],
            systemInstruction: { parts: [{ text: "Eres Patricia Bustos, psicóloga cálida. Ofrece una reflexión terapéutica breve. Invita al usuario a usar su sesión gratuita de cortesía (indicada en la parte inferior derecha) para profundizar." }] }
          })
        });
        const data = await response.json();
        const text = data.candidates?.[0]?.content?.parts?.[0]?.text;
        
        loading.style.display = 'none';
        responseDiv.innerHTML = text.replace(/\n/g, '<br>');
        responseDiv.style.display = 'block';
        return;
      } catch (e) {
        retries++;
        await new Promise(r => setTimeout(r, Math.pow(2, retries) * 1000));
      }
    }
    loading.style.display = 'none';
  }

  function getAIAdvice() {
    const mood = document.getElementById('user-mood').value;
    if (mood.length < 5) return;
    callGemini(`Reflexión para: "${mood}"`);
  }
</script>
